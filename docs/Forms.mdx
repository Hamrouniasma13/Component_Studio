## Form Component

<Playground>
  {() => {
    const formSchema = {
      title: 'A registration form',
      description: 'A simple form example.',
      type: 'object',
      required: ['firstName', 'lastName'],
      properties: {
        firstName: {
          title: 'Surname'
        },
        lastName: {
          title: 'Given name'
        },
        effectiveDate: {
          title: 'Effective date',
          type: 'date'
        },
        hkId: {
          type: 'string',
          title: 'HK ID'
        },
        birth: {
          type: 'date',
          title: 'Birth date'
        },
        gender: {
          title: 'Gender',
          enum: ['Male', 'Female']
        },
        relationship: {
          title: 'Relationship',
          enum: ['Wife', 'Husband']
        },
        email: {
          title: 'Email Address'
        },
        memberClass: {
          title: "Member class",
          enum: ['01', '02', '03']
        },
        comment: {
          title: 'Comment',
        }
      }
    }
    
    const uiSchema = {
      firstName: {
        placeholder: 'Fill text',
      },
      lastName: {
        placeholder: 'Fill text'
      },
      effectiveDate: {
        placeholder: 'YYYY/MM/DD',
      },
      hkId: {
        placeholder: 'Fill numbers',
      },
      birth: {
        widget: 'birthdate'
      },
      relationship: {
        widget: 'select',
        placeholder: 'relationship...'
      },
      email: {
        placeholder: 'xxx@axa.com',
      },
      memberClass: {
        widget: 'select',
      },
      comment: {
        widget: 'textarea',
        placeholder: 'Type something...',
      }
    }
    const validation = values => {
      const errors = {}
      if (values.firstName && values.firstName.length > 10) {
        errors.firstName = 'Message is too long !'
      }

      return errors
    }

    return <Form validation={validation} formSchema={formSchema} uiSchema={uiSchema}/>

}}

</Playground>
